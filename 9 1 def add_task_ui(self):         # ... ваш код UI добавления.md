```python
def add_task(self, title, description, priority=Priority.MEDIUM):
    """–î–æ–±–∞–≤–ª—è–µ—Ç –∑–∞–¥–∞—á—É –≤ –º–µ–Ω–µ–¥–∂–µ—Ä (–¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Ä–µ–∞–ª–∏–∑–æ–≤–∞–Ω –≤ TaskManager)"""
    if not title or not title.strip():
        return None
    
    task = Task(
        id=self.next_id,
        title=title.strip(),
        description=description.strip() if description else "",
        priority=priority
    )
    
    self.tasks.append(task)
    self.next_id += 1
    self.save_tasks()
    return task
```
